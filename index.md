prepTime = ""

tags = [ "Mardi Gras",]
title = "Creole Shrimp Skewers"
ingredients = "* 2 tablespoons Creole Seasoning\n* 6 tablespoons unsalted butter, melted\n* 1 pound medium shrimp (30/40) peeled and deveined"
directions = "* Preheat the oven to 425 degrees F.\n* Combine the Creole Seasoning and melted butter in a medium bowl. Whisk to combine. Place 3 tablespoons of this mixture in a shallow dish and reserve the remaining mixture for later.\n* Skewer the shrimp on small 6 inch skewers, using 2-3 shrimp per skewer. ## Sweet and Spicy Pecans 
+++
categories = [ "Appetizers",]
cookTime = "20 min"
prepTime = "15 min"
servings = "8"
tags = []
title = "Sweet and Spicy Pecans"
ingredients = "* Cooking spray\n* 2 cups pecans, pieces or halves\n* 1 tablespoon unsalted butter\n* 3 tablespoons dark brown sugar\n* 1 1/2 tablespoons maple syrup\n* 1 1/2 teaspoons kosher salt\n* 3/4 teaspoon cayenne pepper\n* 1/2 teaspoon ground cinnamon"
directions = "* Gather the ingredients\n* Position a rack in the center of the oven and heat to 350 F. Prepare a baking sheet by lining it with aluminum foil and spraying the foil with cooking spray.\n* Spread the pecans on the baking sheet and toast them in the oven until dark and fragrant, about 10 minutes. Stir every 3 or 4 minutes.\n* While the pecans are toasting, prepare the coating mixture. Microwave the butter in a large microwave-safe bowl until melted. Stir in the brown sugar, maple syrup, salt, cayenne, and cinnamon.\n* Once the pecans are fragrant, remove them from the oven and pour the hot nuts into the coating mixture, reserving the baking sheet. Be careful as it might splutter and hiss. Stir until the nuts are evenly coated.\n* Pour the nuts back onto the reserved baking sheet, spreading them out into an even layer.\n* Bake for another 10 minutes, stirring after every 3 to 4 minutes. After 10 minutes, remove the nuts from the oven and allow them to cool and crisp up on the baking sheet.\n* Once cool, break the nuts apart or coarsely chop them.\n* Use them in your favorite recipe or simply serve and enjoy!"
notes = "* Originally from: [https://www.thespruceeats.com/sweet-and-spicy-candied-pecans-521158](https://www.thespruceeats.com/sweet-and-spicy-candied-pecans-521158 \"https://www.thespruceeats.com/sweet-and-spicy-candied-pecans-521158\")\n* If you are very sensitive to spices, you might want to reduce the cayenne. On the other hand, if you’re a pepper enthusiast, feel free to increase it. The amount as written gives the nuts a subtle but noticeable “kick.\"\n* If you don't have a microwave, you can melt the butter in a saucepan on the stove. Remove it from the heat before adding and stirring in the brown sugar and spices.\n* You can use other nuts in place of the pecans. Yet, pecans are preferred because their sweet taste goes perfectly with this slightly spicy, slightly savory preparation.\n* Store the sweet and spicy pecans in an airtight container at room temperature. They will retain their flavor and stay good up to a week."
